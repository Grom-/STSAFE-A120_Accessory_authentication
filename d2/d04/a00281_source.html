27 /**
28  * \brief Parse the extensions part of a certificate
29  * \param[in] *extensions pointer to the extensions part of a certificate
30  * \param[in] extensionsSize Size of the extensions field
31  * \param[out] *stse_certificate pointer to the stse_certificate_t that will be filled
32  */
33 static void stse_certificate_parse_extensions(const PLAT_UI8 *extensions, PLAT_I32 extensionsSize, stse_certificate_t *stse_certificate)
34 {
35  const PLAT_UI8 *next_value;
36  PLAT_I32 size, tag, parsed;
37  stse_cert_extension_t ext_st = {.type = -1, .value = NULL, .valueSize = 0};
38 
39  if(extensions != NULL && stse_certificate != NULL)
40  {
41  /* Start parsing the extensions, extensions is a sequence */
42  tag = stse_certificate_identify_ASN1_TLV(extensions, &parsed, &size, &next_value);
43  if (tag == TAG_SEQUENCE)
44  {
45  /* Now we loop over sequences */
46  while (next_value < extensions + extensionsSize)
47  {
48  tag = stse_certificate_identify_ASN1_TLV(next_value, &parsed, &size, &next_value);
49  if (tag == TAG_SEQUENCE)
50  {
51  stse_certificate_parse_extension(next_value, &stse_certificate->extensionsFlags, &ext_st);
52  }
53  next_value += size;
54  }
55  }
56  }
57 }

59 /**
60  * \brief Parse the tbs part of a certificate
61  * \param[in] *tbs pointer to the TBSCertificate field of a certificate
62  * \param[in] *tbsSize size of the TBSCertificate
63  * \param[out] *stse_certificate pointer to the stse_certificate_t that will be filled
64  */
65 static void parsetbsCertificate(const PLAT_UI8 *tbs, PLAT_I32 tbsSize, stse_certificate_t *stse_certificate)
66 {
67  const PLAT_UI8 *next_value;
68  PLAT_I32 size, tag, parsed;
69 
70  if(tbs != NULL && stse_certificate != NULL)
71  {
72  /* Start parsing the TBS, tbs is a sequence */
73  tag = stse_certificate_identify_ASN1_TLV(tbs, &parsed, &size, &next_value);
74  if (tag == TAG_SEQUENCE)
75  {
76  /* Now we expect the version */
77  tag = stse_certificate_identify_ASN1_TLV(next_value, &parsed, &size, &next_value);
78  if (tag == TAG_x509VERSION)
79  {
80  stse_certificate_parse_X509_version(next_value, stse_certificate, &next_value);
81  }
82  else /* The Version is optional, if it's not present then 0 (v1) is assumed */
83  {
84  stse_certificate->x509Version = 0;
85  next_value -= parsed;
86  }
87  /* Move on to Serial Number */
88  stse_certificate_parse_integer(next_value, &stse_certificate->serialNumber, &stse_certificate->serialNumberSize, &next_value);
89  /* Parse tbsSignature(Algorihtm)*/
90  stse_certificate_parse_signature_algorithm(next_value, &stse_certificate->signature, &next_value);
91  /* Now we face the issuer */
92  tag = stse_certificate_identify_ASN1_TLV(next_value, &parsed, &size, &next_value);
93  if (tag == TAG_SEQUENCE)
94  {
95  stse_certificate->issuer = next_value - parsed;
96  stse_certificate->issuerSize = size + parsed;
97  }
98  next_value += size; /* Issuer is not parsed */
99  /* Now we face validity */
100  tag = stse_certificate_identify_ASN1_TLV(next_value, &parsed, &size, &next_value);
101  if (tag == TAG_SEQUENCE)
102  {
103  stse_certificate->validity = next_value - parsed;
104  stse_certificate->validitySize = size + parsed;
105  }
106  next_value += size; /* Validity is not parsed */
107  /* Now we face Subject */
108  tag = stse_certificate_identify_ASN1_TLV(next_value, &parsed, &size, &next_value);
109  if (tag == TAG_SEQUENCE)
110  {
111  stse_certificate->subject = next_value - parsed;
112  stse_certificate->subjectSize = size + parsed;
113  }
114  next_value += size; /* Subject is not parsed */
115  /* Now we face SubjectPublicKey */
116  stse_certificate_parse_ECC_public_key(next_value, stse_certificate, &next_value);
117  /* Now we face Optional Extensions, but those are optional */
118  if (next_value < tbs + tbsSize)
119  {
120  tag = stse_certificate_identify_ASN1_TLV(next_value, &parsed, &size, &next_value);
121  if (tag == TAG_extensions)
122  {
123  stse_certificate->extensions = next_value;
124  stse_certificate->extensionsSize = size;
125  stse_certificate_parse_extensions(stse_certificate->extensions, stse_certificate->extensionsSize, stse_certificate);
126  }
127  }
128  }
129  }

171 stse_ReturnCode_t stse_certificate_parse(const PLAT_UI8 *cert, stse_certificate_t *stse_certificate, const PLAT_UI8 **next)
172 {
173  PLAT_I32 total_size, tag, size, parsed;
174  const PLAT_UI8 *next_thing;
175 
176  stse_certificate_init(stse_certificate);
177  /* First we start by looking at the first item, this will be used to set the size of the whole certificate */
178  /* Then the parsing of the other subfields will start */
179  tag = stse_certificate_identify_ASN1_TLV(cert, &parsed, &total_size, &next_thing);
180  if (tag != TAG_SEQUENCE)
181  {
182  return(STSE_CERT_UNEXPECTED_SEQUENCE);
183  }
184 
185  /* Get tbs */
186  tag = stse_certificate_identify_ASN1_TLV(next_thing, &parsed, &size, &next_thing);
187  if (tag != TAG_SEQUENCE)
188  {
189  return(STSE_CERT_UNEXPECTED_SEQUENCE);
190  }
191  stse_certificate->tbs = next_thing-parsed;
192  stse_certificate->tbsSize = size+parsed;
193 
194  /* Get SignatureAlgorithm */
195  next_thing += size;
196  stse_certificate_parse_signature_algorithm(next_thing, &stse_certificate->SignatureAlgorithm, &next_thing);
197 
198  /* Now we should have the Signature. If it's ECDSA parse it! */
199  if (stse_certificate->SignatureAlgorithm != -1)
200  {
201  stse_certificate_parse_ECDSA_signature(next_thing, stse_certificate, &next_thing);
202  }
203 
204  /* We have completed the parsing of the TOP level */
205  /* Let's move on to parse the tbs */
206  parsetbsCertificate(stse_certificate->tbs, stse_certificate->tbsSize, stse_certificate);
207 
208  if (next != NULL)
209  {
210  *next = next_thing;
211  }
212 
213  return(STSE_OK);
214 }

216 stse_ReturnCode_t stse_certificate_is_valid(const stse_certificate_t *stse_certificate, const stse_cert_validity_t *currentTime)
217 {
218  if (
219  (stse_certificate->issuer == NULL || stse_certificate->issuerSize == -1) ||
220  (stse_certificate->subject == NULL || stse_certificate->subjectSize == -1) ||
221  (stse_certificate->serialNumber == NULL || stse_certificate->serialNumberSize == -1) ||
222  (stse_certificate->signature == -1 || stse_certificate->SignatureAlgorithm != stse_certificate->signature) ||
223  (stse_certificate->validity == NULL || stse_certificate->validitySize == -1) ||
224  (stse_certificate->EllipticCurve == -1 || stse_certificate->x509Version == -1)||
225  (stse_certificate->PubKey.fsize <= 0 || stse_certificate->PubKey.pX == NULL) ||
226  (*stse_certificate->pPubKey_point_representation_id == 0x04 && stse_certificate->PubKey.pY == NULL) ||
227  (*stse_certificate->pPubKey_point_representation_id != 0x04 && stse_certificate->PubKey.pY != NULL) ||
228  (stse_certificate->Sign.pR == NULL || stse_certificate->Sign.rSize == -1) ||
229  (stse_certificate->Sign.pS == NULL || stse_certificate->Sign.sSize == -1)
230  )
231  {
232  return(STSE_CERT_INVALID_CERTIFICATE);
233  }
234 
235  if (currentTime != NULL)
236  {
237  stse_cert_validity_t notBefore_st, notAfter_st;
238  stse_certificate_parse_validity(stse_certificate->validity, &notBefore_st, &notAfter_st, NULL);
239 
240  if (stse_certificate_date_compare(currentTime, &notBefore_st) < 0)
241  {
242  return(STSE_CERT_INVALID_CERTIFICATE);
243  }
244 
245  if (stse_certificate_date_compare(currentTime, &notAfter_st) > 0)
246  {
247  return(STSE_CERT_INVALID_CERTIFICATE);
248  }
249  }
250 
251  return(STSE_OK);
252 }

268 stse_ReturnCode_t stse_certificate_is_parent(const stse_certificate_t *parent, const stse_certificate_t *child, const stse_cert_validity_t *currentTime)
269 {
270  /* Get the two name structure to be compared */
271  const PLAT_UI8 *name1 = child->issuer;
272  PLAT_I32 name1Size = child->issuerSize;
273  const PLAT_UI8 *name2 = parent->subject;
274  PLAT_I32 name2Size = parent->subjectSize;
275  PLAT_I32 found, parsed;
276  const PLAT_UI8 *attr2_loop;
277  const PLAT_UI8 *attr1, *attr2;
278  stse_cert_attribute_t attr1_st, attr2_st;
279 
280  /* Do not compare two invalid certificates */
281  if (stse_certificate_is_valid(parent, currentTime) != STSE_OK)
282  {
283  return(STSE_CERT_INVALID_CERTIFICATE);
284  }
285 
286  if (stse_certificate_is_valid(child, currentTime) != STSE_OK)
287  {
288  return(STSE_CERT_INVALID_CERTIFICATE);
289  }
290 
291  if ( stse_certificate_is_CA(parent) != STSE_OK)
292  {
293  return(STSE_CERT_INVALID_CERTIFICATE);
294  }
295 
296 
297  /* Check that they have the same number of attributes */
298  if (stse_certificate_count_attributes(name1) != stse_certificate_count_attributes(name2))
299  {
300  return (STSE_CERT_CA_NOT_MATCHING);
301  }
302 
303  /* Now we need to scan name1 and look for corresponding attribute in name2 */
304  /* Start entering in the root sequence */
305  if (stse_certificate_identify_ASN1_TLV(name1, &parsed, &found, &name1) != TAG_SEQUENCE)
306  {
307  return(STSE_CERT_INVALID_CERTIFICATE);
308  }
309 
310  if (stse_certificate_identify_ASN1_TLV(name2, &parsed, &found, &name2) != TAG_SEQUENCE)
311  {
312  return(STSE_CERT_INVALID_CERTIFICATE);
313  }
314 
315  /* Loop on all RDN */
316  while ( (name1 < child->issuer + name1Size) && (name2<parent->subject+name2Size) )
317  {
318  /* Parse RDN1 and RDN2. The SEQUENCE is order so this should match */
319  stse_certificate_parse_relative_distinguished_name(name1, &name1, &attr1);
320  stse_certificate_parse_relative_distinguished_name(name2, &name2, &attr2);
321  /* Now loop on the attributes inside the RDN, which are not ordered */
322  while (attr1 < name1) /*Loop untill the next RDN1 */
323  {
324  /* A single DN is a SET which might (but usually doesn't) contain more Attributes */
325  /* Since the SET is not ordered we need to select the attribute from the name1 and scan
326  all the SET of name2 for a matching attribute */
327 
328  stse_certificate_parse_attribute(attr1, &attr1_st, &attr1); /*Read the attribute of Name1 */
329 
330  attr2_loop = attr2; /* copy attr2 as attr2loop will loop for each attribute of RDN1 */
331  found = 0; /* Flag to tell us if we found or not the attribute in Name2 */
332  do /* Scans the whole DN SET in Name2 */
333  {
334  /* Read Attribute */
335  stse_certificate_parse_attribute(attr2_loop, &attr2_st, &attr2_loop);
336  /* Check if it match with the attribute of name 1 */
337  if (attr1_st.type == attr2_st.type)
338  {
339  found = 1; /* Yes, we found the matching attribute */
340 
341  /* Now we need to compare */
342  /* In this if we need to either break the do/while or return -1 */
343  /* We don't support IA5String so we support only BITSTRING, UTF8 and PrintableString */
344 
345  /* Now consider the case where attr1 is a string (either Printable or UTF8) */
346  if (attr1_st.strFormat == TAG_PrintableString || attr1_st.strFormat == TAG_UTF8String)
347  {
348  /* Then the attr2 should be of string type with the same size */
349  if ((attr2_st.strFormat != TAG_PrintableString && attr2_st.strFormat != TAG_UTF8String) ||
350  attr1_st.strSize != attr2_st.strSize)
351  {
352  return (STSE_CERT_CA_NOT_MATCHING);
353  }
354  /* Now they are both string type and of same size, they should match */
355  if (stse_certificate_case_insensitive_compare(attr1_st.str, attr2_st.str, attr1_st.strSize) != 0)
356  {
357  return (STSE_CERT_CA_NOT_MATCHING);
358  }
359  /* Here they are both string type, of same size, and match. So break the loop */
360  break;
361  }
362  /* If we are here we didn't return -1 and we didn't break */
363  /* Therefore the type of attr1 was not among the allowed ones (e.g. it might be IA5String) */
364  /* Just try a very raw comparison.. it's the only thing we can do */
365  /* Note: this perfectly cover the BITSTRING case */
366  /* attributes should be of same type and size */
367  if (attr1_st.strFormat == attr2_st.strFormat && attr1_st.strSize == attr2_st.strSize)
368  {
369  /* And they should match*/
370  if (memcmp(attr1_st.str, attr2_st.str, (PLAT_UI32)attr1_st.strSize) != 0)
371  {
372  return (STSE_CERT_CA_NOT_MATCHING);
373  }
374  /* Here they match, so we break the loop */
375  break;
376  }
377  return (STSE_CERT_CA_NOT_MATCHING);
378  }
379  } while (attr2_loop < name2);
380  /* Did we found the attribute ? If not return -1 */
381  if (found != 1)
382  {
383  return (STSE_CERT_CA_NOT_MATCHING);
384  }
385  }
386  }
387 
388  /* If we are here then certificates are ok. */
389  /* But is the signature valid ? */
390  return(stse_certificate_verify_cert_signature(parent, child));
391 }

393 stse_ReturnCode_t stse_certificate_parse_chain(
394  PLAT_UI8 *rootCA,
395  PLAT_UI8 *certChain,
396  PLAT_UI16 certChainSize,
397  stse_certificate_t *leafCert)
398 {
399  stse_ReturnCode_t ret;
400  PLAT_UI8 *next;
401  PLAT_UI8 *eofCertChain = certChain + certChainSize;
402  stse_certificate_t parsed_parent_cert;
403  stse_certificate_t parsed_child_cert;
404  /* Parse first parent certificate */
405  ret = stse_certificate_parse(certChain, &parsed_child_cert, (const PLAT_UI8 **)&next);
406  if (ret != STSE_OK)
407  {
408  return ret;
409  }
410  /* If a root CA is passed in arguments, process it */
411  if(rootCA != NULL)
412  {
413  /* Parse root CA certificate */
414  ret = stse_certificate_parse(rootCA, &parsed_parent_cert, NULL);
415  if (ret != STSE_OK)
416  {
417  return ret;
418  }
419  /* Verify first intermediate CA against root CA */
420  ret = stse_certificate_is_parent(&parsed_parent_cert, &parsed_child_cert, NULL);
421  if (ret != STSE_OK)
422  {
423  return ret;
424  }
425  }
426  /* Loop over each certificate in the chain */
427  while(next < eofCertChain)
428  {
429  /* Child becomes parent of the next certificate */
430  stse_certificate_copy(&parsed_parent_cert, &parsed_child_cert);
431  /* Parse next child certificate */
432  ret = stse_certificate_parse(next, &parsed_child_cert, (const PLAT_UI8 **)&next);
433  if (ret != STSE_OK)
434  {
435  return ret;
436  }
437  /* Verify child cert signature against parent public key */
438  ret = stse_certificate_is_parent(&parsed_parent_cert, &parsed_child_cert, NULL);
439  if (ret != STSE_OK)
440  {
441  return ret;
442  }
443  }
444  /* If leafCert != NULL, it need to be returned */
445  if(leafCert != NULL)
446  {
447  /* Last child certificate is returned as leaf certificate */
448  stse_certificate_copy(leafCert, &parsed_child_cert);
449  }
450  return STSE_OK;
451 }
