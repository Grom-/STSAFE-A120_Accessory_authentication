70 /**
71  * \brief Check whether "parent" issued "child"
72  * \param[in] parent pointer to the parsed x509 certificate of the supposed issuer of child
73  * \param[in] child pointer to the parsed x509 certificate of the certificated supposedly issued by parent
74  * \param[in] currentTime pointer to a stse_cert_validity_t with the current DateTime. If NULL no date check is done.
75  * \return \ref stse_ReturnCode_t : STSE_OK on success ; error code otherwise
76  * \note RFC 5280 mandates an extremely expensive comparison, which we don't perform.
77  * As result of this simplification some legitimate parentship relation could be not acknowledged (the opposite should not occur).
78  * \warning This check only a single level parent relashion (i.e. whether parent issued child)
79  */
80 stse_ReturnCode_t stse_certificate_is_parent(const stse_certificate_t *parent, const stse_certificate_t *child, const stse_cert_validity_t *currentTime);
