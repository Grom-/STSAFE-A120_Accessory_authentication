20 stse_ReturnCode_t stsafea_cmac_hmac_compute(
21  stse_Handler_t * pSTSE,
22  PLAT_UI8 slot_number,
23  PLAT_UI8 * pMessage,
24  PLAT_UI8 message_length,
25  PLAT_UI8 * pMac,
26  PLAT_UI8 mac_length)
27 {
28  stse_ReturnCode_t ret;
29  PLAT_UI8 cmd_header = STSAFEA_CMD_GENERATE_MAC;
30  PLAT_UI8 sub_command_distinguisher = 0x03;
31  PLAT_UI8 rsp_header;
32  stse_cmd_access_conditions_t cmd_ac_info;
33  PLAT_UI8 cmd_encryption_flag = 0;
34  PLAT_UI8 rsp_encryption_flag = 0;
35 
36  /* - Check stsafe handler initialization */
37 
38  if (pSTSE == NULL)
39  {
40  return( STSE_SERVICE_HANDLER_NOT_INITIALISED );
41  }
42 
43  if((pMac == NULL)
44  || (pMessage == NULL))
45  {
46  return( STSE_SERVICE_INVALID_PARAMETER );
47  }
48 
49  stsafea_perso_info_get_cmd_encrypt_flag(pSTSE->pPerso_info, cmd_header, &cmd_encryption_flag);
50  stsafea_perso_info_get_rsp_encrypt_flag(pSTSE->pPerso_info, cmd_header, &rsp_encryption_flag);
51  stsafea_perso_info_get_cmd_AC(pSTSE->pPerso_info, cmd_header, &cmd_ac_info);
52 
53  /* - Prepare CMD Frame : [HEADER] [CMD DISTINGUISHER] [SLOT] [MAC LENGTH] [MESSAGE] */
54 
55  stse_frame_allocate(CmdFrame);
56  stse_frame_element_allocate_push(&CmdFrame,eCmd_header,STSAFEA_HEADER_SIZE,&cmd_header);
57  stse_frame_element_allocate_push(&CmdFrame,eSub_command_distinguisher,1,&sub_command_distinguisher);
58  stse_frame_element_allocate_push(&CmdFrame,eSlot_number,1,&slot_number);
59  stse_frame_element_allocate_push(&CmdFrame,eMac_length,1,&mac_length);
60  stse_frame_element_allocate_push(&CmdFrame,eMessage,message_length,pMessage);
61 
62  /* - Prepare RSP Frame : [HEADER] [MAC] */
63 
64  stse_frame_allocate(RspFrame);
65  stse_frame_element_allocate_push(&RspFrame,eRsp_header,STSAFEA_HEADER_SIZE,&rsp_header);
66  stse_frame_element_allocate_push(&RspFrame,eMac,mac_length,pMac);
67 
68  if (cmd_encryption_flag || rsp_encryption_flag)
69  {
70  ret = stsafea_session_encrypted_transfer (pSTSE->pActive_host_session,
71  &CmdFrame,
72  &RspFrame,
73  cmd_encryption_flag,
74  rsp_encryption_flag,
75  cmd_ac_info,
76  stsafea_cmd_timings[pSTSE->device_type][cmd_header]
77  );
78  } else if (cmd_ac_info != STSE_CMD_AC_FREE) {
79  ret = stsafea_session_authenticated_transfer( pSTSE->pActive_host_session,
80  &CmdFrame,
81  &RspFrame,
82  cmd_ac_info,
83  stsafea_cmd_timings[pSTSE->device_type][cmd_header]
84  );
85  } else {
86 
87  /* - Perform Transfer*/
88  ret = stse_frame_transfer(pSTSE,
89  &CmdFrame,
90  &RspFrame,
91  stsafea_cmd_timings[pSTSE->device_type][cmd_header]
92  );
93 
94  }
95  return ret;
96 }
