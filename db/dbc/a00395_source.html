182  const PLAT_UI8 *issuer; /*!< Pointer to Issuer */
183  PLAT_I32 issuerSize; /*!< Size of Issuer */
184  const PLAT_UI8 *validity; /*!< Pointer to Validity */
185  PLAT_I32 validitySize; /*!< Size of Validity */
186  const PLAT_UI8 *subject; /*!< Pointer to subject */
187  PLAT_I32 subjectSize; /*!< Size of Subject */
188  PLAT_I32 EllipticCurve; /*!< Identifies the Elliptic Curve */
189  stse_cert_public_key_t PubKey; /*!< Contain the struct \ref stse_cert_public_key_t */
190  const PLAT_UI8 *extensions; /*!< Pointer to Extensions */
191  PLAT_I32 extensionsSize; /*!< Size of Extensions */
192  PLAT_UI32 extensionsFlags; /*!< Integer encoding the fields present in the tcs: \n
193  * bit 0: BasicConstraints is present \n
194  * bit 1: BasicConstraints is critical \n
195  * bit 2: BasicConstraints indicates this is a CA \n
196  * bit 3: BasicConstraints has a pathLenConstraint \n
197  * bit 4-7: PathLen \n
198  * bit 8: keyUsage is present \n
199  * bit 9: keyUsage is critical \n
200  * bit 15: keyUsage field decipherOnly \n
201  * bit 16: keyUsage field digitalSignature \n
202  * bit 17: keyUsage field contentCommitment/nonRepudiation \n
203  * bit 18: keyUsage field keyEncipherment \n
204  * bit 19: keyUsage field dataEncipherment \n
205  * bit 20: keyUsage field keyAgreement \n
206  * bit 21: keyUsage field keyCertSign \n
207  * bit 22: keyUsage field cRLSign \n
208  * bit 23: keyUsage field encipherOnly \n
209  * bit 24: ExtendedKeyUsage is present \n
210  * bit 25: ExtendedKeyUsage is critical \n
211  * bit 27-29: Unused
212  */
213  PLAT_I32 SignatureAlgorithm; /*!< SignatureAlgorithm */
214  stse_cert_signature_t Sign; /*!< Contain the struct \ref stse_cert_signature_t */

222 /** \brief Macro which checks if a TAG is valid (known) */
223 #define IS_VALID_TAG(tag) (( \
224  (TAG_BOOLEAN <= (tag) && (tag) <= TAG_ObjectDescriptor) || \
225  (tag) == TAG_ENUMERATED || \
226  (tag) == TAG_UTF8String || \
227  (tag) == TAG_PrintableString || \
228  (TAG_IA5String <= (tag) && (tag) <= TAG_GeneralizedTime) || \
229  (tag) == TAG_SEQUENCE || \
230  (tag) == TAG_SET || \
231  (TAG_x509VERSION <= (tag) && (tag) <= TAG_extensions) ) ? 1 : 0)
