190 PLAT_I32 stse_certificate_print_bit_string(const PLAT_UI8 *value, PLAT_I32 size)
191 {
192  PLAT_I32 i;
193  /* The first byte of Value identifies the number of discarded bits in the LSB.
194  This is a feature we don't support
195  */
196  if (value[0] != 0)
197  {
198  printf("BITSTRING exclues some LSBits, we don't handle that, error!\n");
199  return(-1);
200  }
201  /* Print the remaining string */
202  for (i = 1; i < size; i++)
203  {
204  printf("%02X", value[i]);
205  if (i % 16 == 0 && i != size - 1)
206  printf("\n\t\t");
207  }
208  return(0);
209 }

211 PLAT_I32 stse_certificate_print_integer(const PLAT_UI8 *value, PLAT_I32 size)
212 {
213  PLAT_I32 i = 0;
214  /* The first byte tells us if it's positive or negative. We don't support negative */
215  if (value[0] >> 7 == 1)
216  {
217  printf("Integer is negative, we don't handle that, error!\n");
218  return(-1);
219  }
220  /* First byte might be zero in case Integer is positive and first byte >127 */
221  /* So skip it */
222  if (value[0] == 0)
223  {
224  i++;
225  }
226  /* And print the rest */
227  for (; i < size; i++)
228  {
229  printf("%02X", value[i]);
230  if (i % 16 == 0 && (i != size - 1) && i != 0)
231  printf("\n\t\t");
232  }
233  return(0);
234 }

247 void stse_certificate_print_parsed_cert(stse_certificate_t *stse_certificate)
248 {
249  //printf("Fields: %08X\n", stse_certificate->fields);
250  //printFields(stse_certificate->fields);
251 
252  printf("\n\r\t x509 Version: %" PRId32, stse_certificate->x509Version + 1);
253  if (stse_certificate->serialNumber != NULL && stse_certificate->serialNumberSize > 0)
254  print_buffer("\n\r\tSerialNumber: ", stse_certificate->serialNumber, stse_certificate->serialNumberSize);
255  if (stse_certificate->issuer != NULL && stse_certificate->issuerSize > 0)
256  {
257  printf("\n\r\t Issuer:");
258  stse_certificate_print_name(stse_certificate->issuer, stse_certificate->issuerSize);
259  }
260  if (stse_certificate->subject != NULL && stse_certificate->subjectSize > 0)
261  {
262  printf("\n\r\t Subject:");
263  stse_certificate_print_name(stse_certificate->subject, stse_certificate->subjectSize);
264  }
265  stse_certificate_print_validity(stse_certificate->validity);
266  printf("\n\r\t SignatureAlgorithm: "); stse_certificate_print_signature_algorithm(stse_certificate->SignatureAlgorithm);
267  printf("\n\r\t tbsSignature: "); stse_certificate_print_signature_algorithm(stse_certificate->signature);
268  printf("\n\r\t EllipticCurve: "); stse_certificate_print_elliptic_curve(stse_certificate->EllipticCurve);
269  printf("\n\r\t Cert PubKey (%s): ", *stse_certificate->pPubKey_point_representation_id == 0x04 ? "Uncompressed" : "Compressed");
270  print_buffer("\n\r\t\t X: ", stse_certificate->PubKey.pX, stse_certificate->PubKey.fsize);
271  if(*stse_certificate->pPubKey_point_representation_id == 0x04)
272  {
273  print_buffer("\n\r\t\t Y: ", stse_certificate->PubKey.pY, stse_certificate->PubKey.fsize);
274  }
275  printf("\n\r\t Cert Signature: ");
276  print_buffer("\n\r\t\t r: ", stse_certificate->Sign.pR, stse_certificate->Sign.rSize);
277  print_buffer("\n\r\t\t s :", stse_certificate->Sign.pS, stse_certificate->Sign.sSize);
278  printExtensions(stse_certificate->extensionsFlags);
279 }

281 void stse_certificate_simple_parse_attribute(const PLAT_UI8 *p, stse_cert_attribute_t *attribute_st, const PLAT_UI8 **next_thing)
282 {
283  PLAT_I32 total_size, size, parsed, tag;
284  const PLAT_UI8 *next = p;
285  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &total_size, &next);
286  total_size += parsed;
287  if (tag == TAG_SET)
288  {
289  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &size, &next);
290  if (tag == TAG_SEQUENCE)
291  {
292  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &size, &next);
293  if (tag == TAG_OBJECT_IDENTIFIER)
294  {
295  attribute_st->type = stse_certificate_identify_attribute(next, size);
296  if (attribute_st->type != -1)
297  {
298  next += size;
299  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &size, &next);
300  attribute_st->strFormat = tag;
301  attribute_st->str = next;
302  attribute_st->strSize = size;
303  }
304  }
305  }
306  }
307  *next_thing = p + total_size;
308 }

310 void stse_certificate_print_name(const PLAT_UI8 *name, PLAT_I32 nameSize)
311 {
312  PLAT_I32 wholesize, tag, parsed;
313  const PLAT_UI8 *next;
314  stse_cert_attribute_t attribute_st;
315  tag = stse_certificate_identify_ASN1_TLV(name, &parsed, &wholesize, &next);
316  if (tag == TAG_SEQUENCE)
317  {
318  /* Here we are inside the name, now loop */
319  while (next < name + nameSize)
320  {
321  attribute_st.type = -1;
322  stse_certificate_simple_parse_attribute(next, &attribute_st, &next);
323  printf("\n\r\t\t "); stse_certificate_print_attribute(attribute_st.type); printf(": ");
324  if (attribute_st.type != -1 &&
325  (attribute_st.strFormat == TAG_PrintableString ||
326  attribute_st.strFormat == TAG_UTF8String)
327  )
328  {
329  PLAT_I32 i;
330  for (i = 0; i < attribute_st.strSize; i++)
331  {
332  printf("%c", attribute_st.str[i]);
333  }
334  }
335  else if (attribute_st.type != -1 && attribute_st.strFormat == TAG_BITSTRING)
336  {
337  stse_certificate_print_bit_string(attribute_st.str, attribute_st.strSize);
338  }
339  }
340  }

353 static void printExtensions(PLAT_UI32 extensionsFlags)
354 {
355  printf("\n\r\t List of Extensions:");
356  if ((extensionsFlags & 1) == 1)
357  {
358  printf("\n\r\t\t BasicConstraints: ");
359  if (((extensionsFlags >> 1) & 1) == 1)
360  {
361  printf("Critical. ");
362  }
363  if (((extensionsFlags >> 2) & 1) == 1)
364  {
365  printf("CA certificate. ");
366  }
367  else
368  {
369  printf("Not a CA certificate. ");
370  }
371  if (((extensionsFlags >> 3) & 1) == 1)
372  {
373  printf("PathSize: %" PRId32, (extensionsFlags >> 4) & 15);
374  }
375  }
376  if (((extensionsFlags >> 8) & 1) == 1)
377  {
378  printf("\n\r\t\t KeyUsage: ");
379  if (((extensionsFlags >> 9) & 1) == 1)
380  {
381  printf("Critical. ");
382  }
383  if (((extensionsFlags >> (16+0)) & 1) == 1)
384  {
385  printf("digitalSignature ");
386  }
387  if (((extensionsFlags >> (16 + 1)) & 1) == 1)
388  {
389  printf("contentCommitment ");
390  }
391  if (((extensionsFlags >> (16 + 2)) & 1) == 1)
392  {
393  printf("keyEncipherment ");
394  }
395  if (((extensionsFlags >> (16 + 3)) & 1) == 1)
396  {
397  printf("dataEncipherment ");
398  }
399  if (((extensionsFlags >> (16 + 4)) & 1) == 1)
400  {
401  printf("keyAgreement ");
402  }
403  if (((extensionsFlags >> (16 + 5)) & 1) == 1)
404  {
405  printf("keyCertSign ");
406  }
407  if (((extensionsFlags >> (16 + 6)) & 1) == 1)
408  {
409  printf("cRLSign ");
410  }
411  if (((extensionsFlags >> (16 + 7)) & 1) == 1)
412  {
413  printf("encipherOnly ");
414  }
415  if (((extensionsFlags >> (15)) & 1) == 1)
416  {
417  printf("decipherOnly ");
418  }
419  }
420 
421  if (((extensionsFlags >> 24) & 1) == 1)
422  {
423  printf("\n\r\t\t ExtendedKeyUsage: ");
424  if (((extensionsFlags >> 25) & 1) == 1)
425  {
426  printf("Critical. ");
427  }
428  }
