36 /* Zone Info Record (ZIR) bitfields definition */
37 #define STSAFEA_ZIR_AC_READ_CR_Pos 7
38 #define STSAFEA_ZIR_AC_READ_Pos 4
39 #define STSAFEA_ZIR_AC_UPDATE_CR_Pos 3
40 #define STSAFEA_ZIR_AC_UPDATE_Pos 0
41 #define STSAFEA_ZIR_AC_READ_CR_Msk 128 /*0b10000000*/
42 #define STSAFEA_ZIR_AC_READ_Msk 112 /*0b01110000*/
43 #define STSAFEA_ZIR_AC_UPDATE_CR_Msk 8 /*0b00001000*/
44 #define STSAFEA_ZIR_AC_UPDATE_Msk 7 /*0b00000111*/
45 #define STSAFEA_ZIR_AC_READ_CR_GET(x) ( ( (x) & STSAFEA_ZIR_AC_READ_CR_Msk) >> STSAFEA_ZIR_AC_READ_CR_Pos )
46 #define STSAFEA_ZIR_AC_READ_GET(x) ( ( (x) & STSAFEA_ZIR_AC_READ_Msk) >> STSAFEA_ZIR_AC_READ_Pos )
47 #define STSAFEA_ZIR_AC_UPDATE_CR_GET(x) ( ( (x) & STSAFEA_ZIR_AC_UPDATE_CR_Msk) >> STSAFEA_ZIR_AC_UPDATE_CR_Pos )
