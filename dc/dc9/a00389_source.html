21 PLAT_I32 stse_certificate_identify_ASN1_TLV(const PLAT_UI8 *asn1, PLAT_I32 *parsed, PLAT_I32 *size, const PLAT_UI8 **value)
22 {
23  /* TLV format has the TAG in the first byte*/
24  PLAT_I32 tag = (PLAT_I32)asn1[0];
25  PLAT_UI32 count, i;
26 
27  *parsed = 0;
28  /* Check if TAG is valid */
29  if (IS_VALID_TAG(tag) == 0)
30  {
31  //printf("Error! Tag %d is not valid\n", tag);
32  return -1;
33  }
34  /*Second byte tells us about the size if it's < 128. Otherwise if bit7 is set, it tell us how many bytes encode the size */
35  if (asn1[1] >> 7 == 1U) /* if bit7 is set, then size is expressed over multiple byte*/
36  {
37  count = (PLAT_UI32)asn1[1] & 127U; /* Read the number of bytes expressing the size */
38  /* Read the size */
39  *size = 0;
40  for (i = 0; i < count; i++)
41  {
42  *size <<= 8U;
43  *size += (PLAT_I32)asn1[2U + i];
44  }
45  }
46  else /* If bit7 is not set, then it is the size */
47  {
48  count = 0;
49  *size = (PLAT_I32)asn1[1];
50  }
51  /* We have written the output size */
52  /* We effectivly write as output both the pointer to value (which follows the size) and the number of bytes we've scanned */
53  *parsed = 2 + (PLAT_I32)count;
54  *value = asn1 + *parsed;
55  /* And we return the tag */
56  return(tag);
57 }

59 void stse_certificate_parse_integer(const PLAT_UI8 *integer, const PLAT_UI8 **outp, PLAT_I32 *outSize, const PLAT_UI8 **next_thing)
60 {
61  PLAT_I32 size=0, tag, parsed=0;
62  const PLAT_UI8 *next;
63 
64  tag = stse_certificate_identify_ASN1_TLV(integer, &parsed, &size, &next);
65  if (tag == TAG_INTEGER)
66  {
67  PLAT_I32 i = 0;
68 
69  /* The first byte tells us if it's positive or negative */
70  /* So skip it */
71  if (next[0] == 0U)
72  {
73  i++;
74  }
75 
76  /* Now we have our integer */
77  *outp = &next[i];
78  *outSize = size - i;
79  }
80  *next_thing = integer + parsed + size;
81 }

102 PLAT_I32 stse_certificate_get_small_integer(const PLAT_UI8 *value, PLAT_I32 size)
103 {
104  PLAT_UI32 retval = 0;
105  PLAT_I32 i = 0;
106  /*Check that it is small enough and that it is positive*/
107  if (value[0] == 0U)
108  {
109  i++;
110  }
111  if ((size - i) > 4 || value[0] >> 7 == 1U)
112  {
113  return(-1);
114  }
115 
116  for (; i < size; i++)
117  {
118  retval <<= 8;
119  retval += value[i];
120  }
121  if (retval < UINT32_MAX)
122  {
123  return((PLAT_I32)retval);
124  }
125  else
126  {
127  return(-1);
128  }

315 PLAT_I32 stse_certificate_case_insensitive_compare(const PLAT_UI8 *p1, const PLAT_UI8 *p2, PLAT_I32 size)
316 {
317  PLAT_I32 i; /* For loop*/
318  PLAT_UI8 d; /* byte difference */
319 
320  /* Try a raw comparison, if it's ok then we are ok */
321  if (memcmp(p1, p2, (PLAT_UI32)size) == 0)
322  {
323  return 0;
324  }
325  /* Otherwise try to be case insensitive */
326  for (i = 0; i < size; i++)
327  {
328  d = p1[i] ^ p2[i];
329  if (d != 0U)
330  {
331  if (d == 32U || ((p1[i] >= (PLAT_UI8)'a' && p1[i] <= (PLAT_UI8)'z') ||
332  (p1[i] >= (PLAT_UI8)'A' && p1[i] <= (PLAT_UI8)'Z')))
333  {
334  continue;
335  }
336  else
337  {
338  return -1;
339  }
340  }
341  }
342  return(0);
343 }

461 PLAT_I32 stse_certificate_date_compare(const stse_cert_validity_t *D1, const stse_cert_validity_t *D2)
462 {
463  PLAT_I32 test = 0;
464  /* First check if year is the same */
465  if (D1->year != D2->year)
466  {
467  /* If it's not then we have our answer as the difference of the two */
468  test = (PLAT_I32)D1->year - (PLAT_I32)D2->year;
469  }
470  /* If years are equal check the months field */
471  if ((test == 0) && (D1->month != D2->month))
472  {
473  test = (PLAT_I32)D1->month - (PLAT_I32)D2->month;
474  }
475  /* And so on... */
476  if ((test == 0) && (D1->days != D2->days))
477  {
478  test = (PLAT_I32)D1->days - (PLAT_I32)D2->days;
479  }
480  if ((test == 0) && (D1->hours != D2->hours))
481  {
482  test = (PLAT_I32)D1->hours - (PLAT_I32)D2->hours;
483  }
484  if ((test == 0) && (D1->minutes != D2->minutes))
485  {
486  test = (PLAT_I32)D1->minutes - (PLAT_I32)D2->minutes;
487  }
488  if ((test == 0) && (D1->minutes != D2->minutes))
489  {
490  test = (PLAT_I32)D1->minutes - (PLAT_I32)D2->minutes;
491  }
492  if ((test == 0) && (D1->seconds != D2->seconds))
493  {
494  test = (PLAT_I32)D1->seconds - (PLAT_I32)D2->seconds;
495  }
496  /* Now we have three cases */
497  /* test < 0 if D1 < D2 */
498  /* test = 0 if D1 = D2 */
499  /* test > 0 if D1 > D2 */
500  if (test < 0)
501  {
502  return (-1);
503  }
504  else
505  {
506  if (test > 0)
507  {
508  return (1);
509  }
510  }
511  return (0);
512 }

515 {
516  PLAT_I32 i, size = 0, parsed, tag, seqSize;
517  const PLAT_UI8 *next = ext;
518 
519  /* we expect the OID */
520  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &size, &next);
521  if (tag == TAG_OBJECT_IDENTIFIER)
522  {
523  ext_st->type = stse_certificate_case_identify_extension(next, size);
524  next += size;
525 
526  switch (ext_st->type)
527  {
528  case EXTENSION_BC:
529  ext_st->critical = 0;
530  *extFlags |= 1U; /* BC is present */
531  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &size, &next);
532  /* Either we get the boolean or the octect string */
533  if (tag == TAG_BOOLEAN)
534  {
535  if (size == 1 && next[0] != 0U)
536  {
537  ext_st->critical = 1;
538  *extFlags |= 1U << 1; /* BC is critical */
539  }
540  next += size;
541  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &size, &next);
542  }
543  if (tag == TAG_OCTETSTRING)
544  {
545  ext_st->value = next;
546  ext_st->valueSize = size;
547 
548  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &seqSize, &next);
549  /* It should start with a sequence */
550  if (tag == TAG_SEQUENCE)
551  {
552  /* Not CA and No Path */
553  if (seqSize > 0) /* There is a CA or PATH */
554  {
555  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &size, &next);
556  /* Boolean and Optional pathLen (integer) */
557  if (tag == TAG_BOOLEAN && next[0] != 0U)
558  {
559  *extFlags |= 1U << 2; /* BC says it's CA */
560  }
561  /* Is there more ? */
562  if (next < ext_st->value + ext_st->valueSize)
563  { /* Yes, then it's the path */
564  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &size, &next);
565  if (tag == TAG_INTEGER)
566  {
567  *extFlags |= 1U << 3; /* Path is present */
568  *extFlags |= ((PLAT_UI32)stse_certificate_get_small_integer(next, size) & 15U) << 4; /* 4-bit encoding for pathLen */
569  }
570  }
571  }
572  }
573  }
574  break;
575  case EXTENSION_KU:
576  ext_st->critical = 0;
577  *extFlags |= 1U << 8; /* KU is present */
578  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &size, &next);
579  /* Either we get the boolean or the octect string */
580  if (tag == TAG_BOOLEAN)
581  {
582  if (size == 1 && next[0] != 0U)
583  {
584  ext_st->critical = 1;
585  *extFlags |= 1U << 9; /* KU is critical */
586  }
587  next += size;
588  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &size, &next);
589  }
590  if (tag == TAG_OCTETSTRING)
591  {
592  ext_st->value = next;
593  ext_st->valueSize = size;
594 
595  tag = stse_certificate_identify_ASN1_TLV(next, &parsed, &seqSize, &next);
596  /* It should start with a sequence */
597  if (tag == TAG_BITSTRING)
598  {
599 #define MAX(a,b) (((a)>(b))?(a):(b))
600  /* This reads the 8 bits (or less) and put it in the right place */
601  for (i = 0; i < MAX((PLAT_I32)next[0],8); i++)
602  {
603  /* Not very readable */
604  *extFlags |= (((PLAT_UI32)next[1] >> (PLAT_UI8)(7U - (PLAT_UI8)i)) & 1U) << (PLAT_UI8)(16U + (PLAT_UI8)i);
605  }
606  if (next[0] == 9U && seqSize == 2)
607  {
608  *extFlags |= (((PLAT_UI32)next[2] >> 7) & 1U) << (15);
609  }
610 #undef MAX
611  }
612  }
613  break;
614  default:
615  break;
616  }
617  }
