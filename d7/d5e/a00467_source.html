58 stse_ReturnCode_t stsafea_start_hash(
59  stse_Handler_t * pSTSE,
60  stse_hash_algorithm_t sha_algorithm,
61  PLAT_UI8* pMessage,
62  PLAT_UI16 message_size
63 )
64 {
65  stse_ReturnCode_t ret;
66  PLAT_UI8 cmd_header[STSAFEA_EXT_HEADER_SIZE] = {STSAFEA_EXTENDED_COMMAND_PREFIX,STSAFEA_EXTENDED_CMD_START_HASH};
67  PLAT_UI8 rsp_header;
68  PLAT_UI16 hash_algo_id_length = STSAFEA_HASH_ALGO_ID_SIZE;
69 
70  if((pSTSE == NULL)||(pMessage == NULL)||(message_size == 0))
71  {
72  return STSE_SERVICE_INVALID_PARAMETER;
73  }
74 
75  /*- Create CMD frame and populate elements */
76  stse_frame_allocate(CmdFrame);
77  stse_frame_element_allocate_push(&CmdFrame,eCmd_header,STSAFEA_EXT_HEADER_SIZE,cmd_header);
78  stse_frame_element_allocate_push(&CmdFrame,eHashAlgo,hash_algo_id_length,(PLAT_UI8*)&stsafea_hash_info_table[sha_algorithm].id);
79  stse_frame_element_allocate_push(&CmdFrame,eMessage,message_size,pMessage);
80 
81  /*- Create Rsp frame and populate elements*/
82  stse_frame_allocate(RspFrame);
83  stse_frame_element_allocate_push(&RspFrame,eRsp_header,1,&rsp_header);
84 
85  /*- Perform Transfer*/
86  ret = stse_frame_transfer(pSTSE,
87  &CmdFrame,
88  &RspFrame,
89  stsafea_extended_cmd_timings[pSTSE->device_type][STSAFEA_EXTENDED_CMD_START_HASH]
90  );
91 
92  return( ret );
93 }

129 stse_ReturnCode_t stsafea_finish_hash(
130  stse_Handler_t * pSTSE,
131  stse_hash_algorithm_t sha_algorithm,
132  PLAT_UI8* pMessage,
133  PLAT_UI16 message_size,
134  PLAT_UI8* pDigest,
135  PLAT_UI16* pDigest_size
136 )
137 {
138 
139  stse_ReturnCode_t ret;
140  PLAT_UI8 cmd_header[STSAFEA_EXT_HEADER_SIZE] = {STSAFEA_EXTENDED_COMMAND_PREFIX,STSAFEA_EXTENDED_CMD_FINISH_HASH};
141  PLAT_UI8 rsp_header;
142  PLAT_UI8 digest_size_array[STSAFEA_GENERIC_LENGTH_SIZE];
143  PLAT_UI16 expected_digest_size = stsafea_hash_info_table[sha_algorithm].length;
144 
145  /*- Verify Parameters */
146  if(pSTSE == NULL
147  || pDigest == NULL
148  || pDigest_size == NULL
149  || (pMessage != NULL && message_size == 0)
150  || (pMessage == NULL && message_size != 0))
151  {
152  return STSE_SERVICE_INVALID_PARAMETER;
153  }
154 
155  /*- Create CMD frame and populate elements */
156  stse_frame_allocate(CmdFrame);
157  stse_frame_element_allocate_push(&CmdFrame,eCmd_header,STSAFEA_EXT_HEADER_SIZE,cmd_header);
158  stse_frame_element_allocate_push(&CmdFrame,eMessage,message_size,pMessage);
159 
160  /*- Create Rsp frame and populate elements */
161  stse_frame_allocate(RspFrame);
162  stse_frame_element_allocate_push(&RspFrame,eRsp_header,STSAFEA_HEADER_SIZE,&rsp_header);
163  stse_frame_element_allocate_push(&RspFrame,eDigestSize,STSAFEA_GENERIC_LENGTH_SIZE,digest_size_array);
164  stse_frame_element_allocate_push(&RspFrame,eDigest,expected_digest_size,pDigest);
165 
166  /*- Perform Transfer */
167  ret = stse_frame_transfer(pSTSE,
168  &CmdFrame,
169  &RspFrame,
170  stsafea_extended_cmd_timings[pSTSE->device_type][STSAFEA_EXTENDED_CMD_FINISH_HASH]
171  );
172 
173  if(ret == STSE_OK)
174  {
175  *pDigest_size = ARRAY_2B_SWAP_TO_UI16(digest_size_array);
176  *pDigest_size = expected_digest_size;
177  }
178  else
179  {
180  *pDigest_size = 0;
181  }
182 
183  return( ret );
184 }
