25  PLAT_UI8 password_length,
26  PLAT_UI8 * pVerification_status,
27  PLAT_UI8 * pRemaining_tries)
28 {
29  stse_ReturnCode_t ret;
30  PLAT_UI8 cmd_header = STSAFEA_CMD_VERIFY_PASSWORD;
31  PLAT_UI8 rsp_header;
32 
33  if((pSTSE == NULL) || (password_length != STSAFEA_PASSWORD_LENGTH))
34  {
35  return STSE_SERVICE_INVALID_PARAMETER;
36  }
37 
38  /*- Create CMD frame and populate elements */
39  stse_frame_allocate(CmdFrame);
40  stse_frame_element_allocate_push(&CmdFrame,eCmd_header,1,&cmd_header);
41  stse_frame_element_allocate_push(&CmdFrame,ePassword,password_length,pPassword_buffer);
42 
43  /*- Create Rsp frame and populate elements*/
44  stse_frame_allocate(RspFrame);
45  stse_frame_element_allocate_push(&RspFrame,eRsp_header,1,&rsp_header);
46  stse_frame_element_allocate_push(&RspFrame,eVerStat,1,pVerification_status);
47  stse_frame_element_allocate_push(&RspFrame,eRemTri,1,pRemaining_tries);
48 
49  /*- Perform Transfer*/
50  ret = stse_frame_transfer(pSTSE,
51  &CmdFrame,
52  &RspFrame,
53  stsafea_cmd_timings[pSTSE->device_type][STSAFEA_CMD_VERIFY_PASSWORD]
54  );
55 
56  return( ret );
57 }
