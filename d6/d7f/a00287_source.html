22 stse_ReturnCode_t stse_certificate_verify_cert_signature(const stse_certificate_t *parent, const stse_certificate_t *child)
23 {
24  stse_ReturnCode_t ret;
25  stse_hash_algorithm_t hash_algo = stse_certificate_get_sig_hash_algo(child);
26 
27  PLAT_UI32 digestSize = stsafea_hash_info_table[hash_algo].length;
28  PLAT_UI8 digest[digestSize];
29 
30  /*
31  * Only STSAFE-A120 support Hash features
32  */
33  if(stsafe_x509_parser_companion_handler != NULL
34  && stsafe_x509_parser_companion_handler->device_type == STSAFE_A120
35  && hash_algo >= STSE_SHA_256)
36  {
37  ret = stse_compute_hash(
38  stsafe_x509_parser_companion_handler,
39  hash_algo,
40  (PLAT_UI8*)child->tbs,
41  child->tbsSize,
42  digest,
43  (PLAT_UI16*)&digestSize);
44  }
45  else
46  {
47  ret = stse_platform_hash_compute(
48  hash_algo,
49  (PLAT_UI8*)child->tbs,
50  child->tbsSize,
51  digest,
52  &digestSize);
53  }
54 
55  if(ret != STSE_OK)
56  {
57  return (ret);
58  }
59 
60  return(stse_certificate_verify_signature(parent, digest, digestSize, child->Sign.pR, child->Sign.rSize, child->Sign.pS, child->Sign.sSize));
61 }

63 stse_ReturnCode_t stse_certificate_verify_signature(const stse_certificate_t *cert,
64  const PLAT_UI8 *digest,
65  PLAT_I32 digestSize,
66  const PLAT_UI8 *signatureR,
67  PLAT_I32 signatureRsize,
68  const PLAT_UI8 *signatureS,
69  PLAT_I32 signatureSsize)
70 {
71  (void)signatureRsize;
72  (void)signatureSsize;
73  stse_ReturnCode_t ret;
74  stse_ecc_key_type_t key_type = stse_certificate_get_key_type(cert);
75  PLAT_UI8 pub_key_size = stsafea_ecc_info_table[key_type].public_key_size;
76  PLAT_UI8 pub_key[pub_key_size];
77  PLAT_UI8 signature_size = stsafea_ecc_info_table[key_type].signature_size;
78  PLAT_UI8 signature[signature_size];
79  PLAT_UI8 signature_validity;
80 
81  /* Extract and format the public key from the certificate */
82  memcpy(pub_key, cert->PubKey.pX, (pub_key_size>>1));
83  if(*cert->pPubKey_point_representation_id == 0x04)
84  {
85  memcpy(pub_key + (pub_key_size>>1), cert->PubKey.pY, (pub_key_size>>1));
86  }
87  else
88  {
89  if(stsafe_x509_parser_companion_handler != NULL)
90  {
91  stsafea_ecc_decompress_public_key(
92  stsafe_x509_parser_companion_handler,
93  key_type,
94  *cert->pPubKey_point_representation_id,
95  pub_key,
96  pub_key + (pub_key_size>>1));
97  }
98  else
99  {
100  return STSE_CERT_UNSUPPORTED_FEATURE;
101  }
102  }
103 
104  /* Format the signature */
105  memcpy(signature, signatureR, (signature_size>>1));
106  memcpy(signature + (signature_size>>1), signatureS, (signature_size>>1));
107 
108  if(stsafe_x509_parser_companion_handler != NULL)
109  {
110  /* Verify the signature using STSAFE */
111  ret = stse_ecc_verify_signature(
112  stsafe_x509_parser_companion_handler,
113  key_type,
114  pub_key,
115  signature,
116  (PLAT_UI8*)digest,
117  (PLAT_UI16)digestSize,
118  1, /* Message is hashed */
119  &signature_validity);
120 
121  if((ret == STSE_OK) && (signature_validity == 1))
122  {
123  return (STSE_OK);
124  }
125  }
126  else
127  {
128  ret = stse_platform_ecc_verify(
129  key_type,
130  pub_key,
131  (PLAT_UI8*)digest,
132  (PLAT_UI16)digestSize,
133  signature);
134 
135  if(ret == STSE_OK)
136  {
137  return (STSE_OK);
138  }
139  }
140 
141  return (STSE_CERT_INVALID_SIGNATURE);
142 }

144 stse_ecc_key_type_t stse_certificate_get_key_type(const stse_certificate_t *cert)
145 {
146  switch (cert->EllipticCurve)
147  {
148  case EC_P256:
149  return STSE_ECC_KT_NIST_P_256;
150  case EC_P384:
151  return STSE_ECC_KT_NIST_P_384;
152  case EC_P521:
153  return STSE_ECC_KT_NIST_P_521;
154  case EC_bp256r1:
155  return STSE_ECC_KT_BP_P_256;
156  case EC_bp384r1:
157  return STSE_ECC_KT_BP_P_384;
158  case EC_bp512r1:
159  return STSE_ECC_KT_BP_P_512;
160  default:
161  return STSE_ECC_KT_INVALID;
162  }
163 }

166 stse_hash_algorithm_t stse_certificate_get_sig_hash_algo(const stse_certificate_t *cert)
167 {
168  switch (cert->SignatureAlgorithm)
169  {
170  case SIG_ECDSA_SHA1:
171  return STSE_SHA_1;
172  case SIG_ECDSA_SHA224:
173  return STSE_SHA_224;
174  case SIG_ECDSA_SHA256:
175  return STSE_SHA_256;
176  case SIG_ECDSA_SHA384:
177  return STSE_SHA_384;
178  case SIG_ECDSA_SHA512:
179  return STSE_SHA_512;
180  default:
181  return STSE_SHA_INVALID;
182  }
