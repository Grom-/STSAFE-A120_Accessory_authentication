28 stse_ReturnCode_t stse_init(stse_Handler_t *pSTSE)
29 {
30  stse_ReturnCode_t ret = STSE_API_INVALID_PARAMETER;
31 
32  if(pSTSE != NULL)
33  {
34  switch(pSTSE->io.BusType)
35  {
36  case STSE_BUS_TYPE_I2C :
37  ret = stse_platform_i2c_init(pSTSE->io.Busaddr);
38  if(ret != STSE_OK)
39  {
40  return ret;
41  }
42  break;
43 
44  default :
45  return (STSE_CORE_INVALID_PARAMETER);
46  }
47 
48  if(ret != STSE_OK)
49  {
50  return ret;
51  }
52 
53  /* - Initialize Host platform */
54  ret = stse_services_platform_init();
55  if(ret != STSE_OK)
56  {
57  return ret;
58  }
59  ret = stse_crypto_platform_init();
60  if(ret != STSE_OK)
61  {
62  return ret;
63  }
64 
65  /*- STSE device power-on delay */
66  stse_platform_Delay_ms(stsafea_boot_time[pSTSE->device_type]);
67 
68  if (pSTSE->pPerso_info == NULL)
69  {
70  pSTSE->pPerso_info = &dynamic_product_perso;
71  ret = stsafea_perso_info_update(pSTSE);
72  }
73 
74  }
75 
76  return ret;
77 }

85 stse_ReturnCode_t stse_device_power_on(stse_Handler_t * pSTSE)
86 {
87  /* - Check STSAFE handler initialization */
88  if (pSTSE == NULL)
89  {
90  return( STSE_API_INVALID_PARAMETER );
91  }
92 
93  /* - Check STSAFE PowerLineOn callback initialization */
94  if (pSTSE->io.PowerLineOn == NULL)
95  {
96  return( STSE_API_INVALID_PARAMETER );
97  }
98 
99  /* - Power-on the device */
100  pSTSE->io.PowerLineOn(pSTSE->io.Busaddr,pSTSE->io.Devaddr);
101 
102  /* - Wait for device to boot (tboot) */
103  stse_platform_Delay_ms(stsafea_boot_time[pSTSE->device_type]);
104  return( STSE_OK );
105 }

131 stse_ReturnCode_t stse_device_lock(stse_Handler_t *pSTSE, PLAT_UI8 *pPassword, PLAT_UI8 password_length)
132 {
133  stse_ReturnCode_t ret;
134  PLAT_UI8 password_verification_status = 0;
135 
136  if(pSTSE == NULL)
137  {
138  return STSE_API_INVALID_PARAMETER;
139  }
140 
141  if(pPassword == NULL || password_length != STSAFEA_PASSWORD_LENGTH)
142  {
143  return STSE_API_INVALID_PARAMETER;
144  }
145 
146  /*- Password submission */
147  ret = stsafea_verify_password(pSTSE,
148  pPassword,
149  password_length,
150  &password_verification_status,
151  NULL);
152 
153  if(ret != STSE_OK)
154  {
155  return ret;
156  }
157 
158  if(password_verification_status == 0)
159  {
160  return STSE_API_INVALID_PARAMETER;
161  }
162 
163  /* - Switch device Life-cycle to Lock */
164  ret = stsafea_put_life_cyle_state(pSTSE,STSAFEA_LCS_OPERATIONAL_AND_LOCKED);
165 
166  return ret;
167 }

169 stse_ReturnCode_t stse_device_unlock(stse_Handler_t *pSTSE, PLAT_UI8 *pPassword, PLAT_UI8 password_length)
170 {
171  stse_ReturnCode_t ret;
172  PLAT_UI8 password_verification_status = 0;
173  PLAT_UI8 remaining_tries;
174 
175  if(pSTSE == NULL)
176  {
177  return STSE_API_INVALID_PARAMETER;
178  }
179 
180  if(pPassword == NULL || password_length != STSAFEA_PASSWORD_LENGTH)
181  {
182  return STSE_API_INVALID_PARAMETER;
183  }
184 
185  /*- Password submission */
186  ret = stsafea_verify_password(pSTSE,
187  pPassword,
188  password_length,
189  &password_verification_status,
190  &remaining_tries);
191 
192  if(ret != STSE_OK)
193  {
194  return ret;
195  }
196 
197  if(password_verification_status == 0)
198  {
199  return STSE_API_INVALID_PARAMETER;
200  }
201 
202  /* - Switch device Life-cycle to operational */
203  ret = stsafea_put_life_cyle_state(pSTSE, STSAFEA_LCS_OPERATIONAL);
204 
205  return ret;
206 }
