278 static stse_ReturnCode_t stsafea_session_frame_c_mac_compute( stse_session_t *pSession,
279  stse_frame_t *pCmd_frame,
280  PLAT_UI8 *pMAC)
281 {
282  PLAT_UI8 aes_cmac_block[STSAFEA_HOST_AES_BLOCK_SIZE];
283  PLAT_UI8 mac_output_length;
284  PLAT_UI8 mac_type = 0x00;
285  stse_frame_element_t* pElement;
286  PLAT_UI8 aes_block_idx = 0;
287  PLAT_UI16 i ;
288  PLAT_UI16 cmd_payload_length = pCmd_frame->length - pCmd_frame->first_element->length;
289  stse_ReturnCode_t ret = STSE_CORE_INVALID_PARAMETER;
290 
291  if((pSession == NULL)
292  || (pCmd_frame == NULL)
293  || (pMAC == NULL))
294  {
295  return STSE_SERVICE_SESSION_ERROR;
296  }
297 
298  /*- create C-MAC Frame : [0x00] [CMD HEADER] [CMD PAYLOAD LENGTH] [CMD PAYLOAD] */
299  stse_frame_allocate(c_mac_frame);
300  stse_frame_element_allocate_push(&c_mac_frame,eMACType,1,&mac_type);
301  stse_frame_element_allocate_push(&c_mac_frame,
302  eCMD_HEADER,
303  pCmd_frame->first_element->length,
304  pCmd_frame->first_element->pData);
305  stse_frame_element_allocate_push(&c_mac_frame,
306  eCmdPayloadLength,
307  STSAFEA_CMD_RSP_LEN_SIZE,
308  (PLAT_UI8*)&cmd_payload_length);
309  stse_frame_element_swap_byte_order(&eCmdPayloadLength);
310  eCmdPayloadLength.next = pCmd_frame->first_element->next;
311  stse_frame_update(&c_mac_frame);
312 
313  /*- Initialize AES C-MAC computation */
314 
315  ret = stse_platform_aes_cmac_init( pSession->context.host.pHost_MAC_key,
316  (pSession->context.host.key_type == STSE_AES_128_KT)? STSE_AES_128_KEY_SIZE:STSE_AES_256_KEY_SIZE,
317  STSAFEA_MAC_SIZE);
318  if(ret != STSE_OK)
319  {
320  return ret;
321  }
322 
323  /*- Perform First AES-CMAC round with MAC subject info */
324  if(pSession->context.host.pSTSE->device_type == STSAFE_A120)
325  {
326  aes_cmac_block[0] = UI32_B3(pSession->context.host.MAC_counter);
327  aes_cmac_block[1] = UI32_B2(pSession->context.host.MAC_counter);
328  aes_cmac_block[2] = UI32_B1(pSession->context.host.MAC_counter);
329  aes_cmac_block[3] = UI32_B0(pSession->context.host.MAC_counter);
330  aes_cmac_block[4] = 0x00U; /* Subject : Host C-MAC */
331  aes_cmac_block[5] = 0x80U; /* First byte of padding */
332  for (i = 6 ; i < STSAFEA_HOST_AES_BLOCK_SIZE ; i++)
333  {
334  aes_cmac_block[i] = 0x00U; /* 0x00 padding */
335  }
336  } else {
337  aes_cmac_block[0] = UI32_B2(pSession->context.host.MAC_counter);
338  aes_cmac_block[1] = UI32_B1(pSession->context.host.MAC_counter);
339  aes_cmac_block[2] = UI32_B0(pSession->context.host.MAC_counter);
340  aes_cmac_block[3] = 0x00U; /* Subject : Host C-MAC */
341  aes_cmac_block[4] = 0x80U; /* First byte of padding */
342  for (i = 5 ; i < STSAFEA_HOST_AES_BLOCK_SIZE ; i++)
343  {
344  aes_cmac_block[i] = 0x00U; /* 0x00 padding */
345  }
346  }
347 
348 
349  ret = stse_platform_aes_cmac_append(aes_cmac_block, STSAFEA_HOST_AES_BLOCK_SIZE);
350  if(ret != STSE_OK)
351  {
352  return ret;
353  }
354 
355  pElement = c_mac_frame.first_element;
356 
357  /*- Perform additional AES-CMAC round(s) for frame to Authenticate */
358  while(pElement != NULL)
359  {
360  for(i=0;i<pElement->length;i++)
361  {
362  if (aes_block_idx == STSAFEA_HOST_AES_BLOCK_SIZE)
363  {
364  stse_platform_aes_cmac_append(aes_cmac_block, STSAFEA_HOST_AES_BLOCK_SIZE);
365  aes_block_idx = 0;
366  }
367  aes_cmac_block[aes_block_idx]= *(pElement->pData+i);
368  aes_block_idx++;
369  }
370  pElement = pElement->next;
371  }
372  if(aes_block_idx != 0)
373  {
374  ret = stse_platform_aes_cmac_append(aes_cmac_block, aes_block_idx);
375  if(ret != STSE_OK)
376  {
377  return ret;
378  }
379  }
380 
381  /*- Finish AES MAC computation */
382  ret = stse_platform_aes_cmac_compute_finish(aes_cmac_block,&mac_output_length);
383  if(ret != STSE_OK)
384  {
385  return ret;
386  } else if ( mac_output_length != STSAFEA_MAC_SIZE ) {
387  return STSE_CORE_SESSION_ERROR ;
388  }
389  memcpy(pMAC, aes_cmac_block, STSAFEA_MAC_SIZE);
390 
391  return ret;
392 }
