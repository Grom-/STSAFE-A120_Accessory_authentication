71 stse_ReturnCode_t stse_get_ecc_key_slot_info(
72  stse_Handler_t * pSTSE,
73  PLAT_UI8 private_key_slot_number,
74  PLAT_UI8 * pChange_right,
75  PLAT_UI16 * pGlobal_usage_limit,
76  stsafea_private_key_slot_information_t * private_key_slot_info)
77 {
78  stse_ReturnCode_t ret;
79 
80  if(pSTSE == NULL)
81  {
82  return( STSE_API_HANDLER_NOT_INITIALISED );
83  }
84 
85  if(private_key_slot_info == NULL)
86  {
87  return( STSE_API_INVALID_PARAMETER );
88  }
89 
90  PLAT_UI8 private_key_slot_count;
91  PLAT_UI8 slot_info_index;
92 
93  ret = stsafea_query_private_key_slots_count(pSTSE, &private_key_slot_count);
94 
95  stsafea_private_key_slot_information_t private_key_table_info[private_key_slot_count];
96 
97  ret = stsafea_query_private_key_table(pSTSE, private_key_slot_count, pChange_right, pGlobal_usage_limit, private_key_table_info);
98 
99  if(ret != STSE_OK)
100  {
101  return(ret);
102  }
103 
104  for(slot_info_index=0; slot_info_index<private_key_slot_count; slot_info_index++)
105  {
106  if(private_key_table_info[slot_info_index].slot_number == private_key_slot_number)
107  {
108  memcpy(private_key_slot_info, &private_key_table_info[slot_info_index], sizeof(stsafea_private_key_slot_information_t));
109  break;
110  }
111  }
112 
113  if(slot_info_index == private_key_slot_count)
114  {
115  return(STSE_API_KEY_NOT_FOUND);
116  }
117 
118  return ret;
119 }
